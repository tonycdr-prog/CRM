// client/src/lib/offlineQueue.ts
type QueueItem =
  | {
      id: string;
      createdAt: number;
      inspectionId: string;
      type: "responses";
      payload: any; // { responses: [...] }
    }
  | {
      id: string;
      createdAt: number;
      inspectionId: string;
      rowId: string;
      type: "attachment";
      fileName: string;
      mimeType: string;
      sizeBytes: number;
      blob: Blob;
    };

const DB_NAME = "lso_offline";
const DB_VERSION = 2;
const STORE = "queue";

function uid() {
  return `${Date.now()}_${Math.random().toString(36).slice(2, 10)}`;
}

function openDb(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onerror = () => reject(req.error);
    req.onupgradeneeded = () => {
      const db = req.result;
      if (!db.objectStoreNames.contains(STORE)) {
        const store = db.createObjectStore(STORE, { keyPath: "id" });
        store.createIndex("byCreatedAt", "createdAt", { unique: false });
        store.createIndex("byInspectionId", "inspectionId", { unique: false });
        store.createIndex("byInspectionRow", ["inspectionId", "rowId"], { unique: false });
        store.createIndex("byType", "type", { unique: false });
      } else {
        // ensure indexes exist (upgrade safety)
        const store = req.transaction!.objectStore(STORE);
        if (!store.indexNames.contains("byCreatedAt")) store.createIndex("byCreatedAt", "createdAt", { unique: false });
        if (!store.indexNames.contains("byInspectionId")) store.createIndex("byInspectionId", "inspectionId", { unique: false });
        if (!store.indexNames.contains("byInspectionRow"))
          store.createIndex("byInspectionRow", ["inspectionId", "rowId"], { unique: false });
        if (!store.indexNames.contains("byType")) store.createIndex("byType", "type", { unique: false });
      }
    };
    req.onsuccess = () => resolve(req.result);
  });
}

async function addItem(item: QueueItem) {
  const db = await openDb();
  await new Promise<void>((resolve, reject) => {
    const tx = db.transaction(STORE, "readwrite");
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
    tx.objectStore(STORE).add(item);
  });
  db.close();
}

export async function enqueueResponses(inspectionId: string, payload: any) {
  await addItem({
    id: uid(),
    createdAt: Date.now(),
    inspectionId,
    type: "responses",
    payload,
  });
}

export async function enqueueAttachment(args: {
  inspectionId: string;
  rowId: string;
  file: File;
}) {
  await addItem({
    id: uid(),
    createdAt: Date.now(),
    inspectionId: args.inspectionId,
    rowId: args.rowId,
    type: "attachment",
    fileName: args.file.name,
    mimeType: args.file.type || "application/octet-stream",
    sizeBytes: args.file.size,
    blob: args.file,
  });
}

export async function getPendingCount(inspectionId?: string) {
  const db = await openDb();
  const count = await new Promise<number>((resolve, reject) => {
    const tx = db.transaction(STORE, "readonly");
    const store = tx.objectStore(STORE);
    const req = inspectionId ? store.index("byInspectionId").count(inspectionId) : store.count();
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
  db.close();
  return count;
}

export async function getPendingAttachmentCountForRow(inspectionId: string, rowId: string) {
  const db = await openDb();
  const count = await new Promise<number>((resolve, reject) => {
    const tx = db.transaction(STORE, "readonly");
    const idx = tx.objectStore(STORE).index("byInspectionRow");
    const req = idx.count([inspectionId, rowId]);
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });
  db.close();
  return count;
}

export async function listPendingAttachmentsForRow(inspectionId: string, rowId: string) {
  const db = await openDb();
  const items = await new Promise<QueueItem[]>((resolve, reject) => {
    const tx = db.transaction(STORE, "readonly");
    const idx = tx.objectStore(STORE).index("byInspectionRow");
    const req = idx.getAll([inspectionId, rowId]);
    req.onsuccess = () => resolve(req.result as QueueItem[]);
    req.onerror = () => reject(req.error);
  });
  db.close();

  return items
    .filter((i): i is Extract<QueueItem, { type: "attachment" }> => (i as any).type === "attachment")
    .sort((a, b) => a.createdAt - b.createdAt);
}

async function listQueueItems(): Promise<QueueItem[]> {
  const db = await openDb();
  const items = await new Promise<QueueItem[]>((resolve, reject) => {
    const tx = db.transaction(STORE, "readonly");
    const store = tx.objectStore(STORE);
    const idx = store.index("byCreatedAt");
    const req = idx.getAll();
    req.onsuccess = () => resolve(req.result as QueueItem[]);
    req.onerror = () => reject(req.error);
  });
  db.close();
  return items.sort((a, b) => a.createdAt - b.createdAt);
}

async function removeItem(id: string) {
  const db = await openDb();
  await new Promise<void>((resolve, reject) => {
    const tx = db.transaction(STORE, "readwrite");
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
    tx.objectStore(STORE).delete(id);
  });
  db.close();
}

export async function flushQueue(options?: {
  onProgress?: (sent: number, remaining: number) => void;
}) {
  const items = await listQueueItems();
  let sent = 0;

  for (let i = 0; i < items.length; i++) {
    const item = items[i];

    try {
      if (item.type === "responses") {
        const res = await fetch(`/api/inspections/${encodeURIComponent(item.inspectionId)}/responses`, {
          method: "POST",
          credentials: "include",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(item.payload),
        });

        if (res.status === 409) {
          await removeItem(item.id);
          continue;
        }
        if (!res.ok) throw new Error(`Sync failed (${res.status})`);
      }

      if (item.type === "attachment") {
        const fd = new FormData();
        // recreate File from blob so server sees name/type
        const f = new File([item.blob], item.fileName, { type: item.mimeType });
        fd.append("file", f);

        const res = await fetch(
          `/api/inspections/${encodeURIComponent(item.inspectionId)}/rows/${encodeURIComponent(item.rowId)}/attachments`,
          { method: "POST", credentials: "include", body: fd }
        );

        if (res.status === 409) {
          // inspection completed; drop queued upload
          await removeItem(item.id);
          continue;
        }
        if (!res.ok) throw new Error(`Upload sync failed (${res.status})`);
      }

      await removeItem(item.id);
      sent++;
      options?.onProgress?.(sent, items.length - (i + 1));
    } catch {
      break;
    }
  }
}
